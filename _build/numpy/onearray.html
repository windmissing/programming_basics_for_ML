标准化 

dataSet = np.array([[1,2,3],
                   [2,4,6],
                   [3,6,9],
                   [4,8,9]])

def autoNorm(dataSet):
    max = np.max(dataSet, axis=0)
    min = np.min(dataSet, axis=0)
    return ((dataSet -min) / (max-min))

autoNorm(dataSet)

求矩阵的协方差 

import numpy as np
A = np.array([[1,2,3,4],
              [8,7,6,5]])
np.cov(A)   # 2*4 -> 2*2

array([[ 1.66666667, -1.66666667],
       [-1.66666667,  1.66666667]])
np.cov(A.T)   # 4*2 -> 4*4

array([[24.5, 17.5, 10.5,  3.5],
       [17.5, 12.5,  7.5,  2.5],
       [10.5,  7.5,  4.5,  1.5],
       [ 3.5,  2.5,  1.5,  0.5]])
np.cov(A.T, rowvar=0)  # 效果和np.cov(A)一样

array([[ 1.66666667, -1.66666667],
       [-1.66666667,  1.66666667]])
